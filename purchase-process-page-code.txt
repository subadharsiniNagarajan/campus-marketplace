  <script>
    requireAuth();
    var user = getUser();
    document.getElementById('nav-username').textContent = user.name;
    document.getElementById('avatar-initials').textContent = user.name.charAt(0).toUpperCase();

    var API_BASE = (typeof API !== 'undefined') ? API : 'http://localhost:3000';
    var purchaseId = new URLSearchParams(window.location.search).get('id');
    var currentPurchase = null;
    var currentItem = null;

    if (!purchaseId) {
      alert('No purchase ID provided');
      window.location.href = 'buy.html';
    } else {
      loadPurchaseStatus();
      // Auto-refresh every 10 seconds
      setInterval(loadPurchaseStatus, 10000);
    }

    async function loadPurchaseStatus() {
      try {
        // Get purchase details
        var res = await fetch(API_BASE + '/api/purchases/buyer?buyerEmail=' + encodeURIComponent(user.email));
        var data = await res.json();
        
        if (!data.success) {
          throw new Error(data.error || 'Failed to load purchase');
        }
        
        // Find this specific purchase
        var purchase = data.purchases.find(function(p) { return p._id === purchaseId; });
        if (!purchase) {
          throw new Error('Purchase not found');
        }
        
        currentPurchase = purchase;
        
        // Get item details
        var itemRes = await fetch(API_BASE + '/items/' + purchase.itemId);
        var itemData = await itemRes.json();
        if (itemData.success && itemData.item) {
          currentItem = itemData.item;
        }
        
        renderPurchaseStatus(purchase);
        
        document.getElementById('loading-state').style.display = 'none';
        document.getElementById('content-state').style.display = 'block';
      } catch (err) {
        console.error('Load error:', err);
        document.getElementById('loading-state').innerHTML = 
          '<div class="status-icon">⚠️</div>' +
          '<h3>Error Loading Purchase</h3>' +
          '<p style="color:var(--text2);">' + err.message + '</p>' +
          '<button class="btn btn-primary" onclick="window.location.href=\'buy.html\'">Back to Browse</button>';
      }
    }

    function renderPurchaseStatus(purchase) {
      var status = purchase.purchaseStatus;
      
      // Update item info
      if (currentItem) {
        var imgSrc = currentItem.images && currentItem.images[0] 
          ? API_BASE + '/uploads/' + currentItem.images[0]
          : (currentItem.image ? API_BASE + '/uploads/' + currentItem.image : '');
        document.getElementById('item-image').src = imgSrc;
        document.getElementById('item-name').textContent = currentItem.name;
        document.getElementById('item-price').textContent = '₹' + Number(currentItem.price).toLocaleString();
      }
      
      // Update progress steps
      var step1 = document.getElementById('step-1');
      var step2 = document.getElementById('step-2');
      var step3 = document.getElementById('step-3');
      var progressFill = document.getElementById('progress-fill');
      
      // Reset
      step1.classList.remove('active', 'completed');
      step2.classList.remove('active', 'completed');
      step3.classList.remove('active', 'completed');
      
      if (status === 'pending') {
        step1.classList.add('completed');
        step2.classList.add('active');
        progressFill.style.width = '50%';
        
        document.getElementById('status-icon').textContent = '⏳';
        document.getElementById('status-title').textContent = 'Waiting for Seller Approval';
        document.getElementById('status-message').textContent = 
          'Your purchase request has been sent to the seller. They will review and respond soon.';
      } else if (status === 'accepted' || status === 'reserved') {
        step1.classList.add('completed');
        step2.classList.add('completed');
        step3.classList.add('completed');
        progressFill.style.width = '100%';
        
        document.getElementById('status-icon').textContent = '🎉';
        document.getElementById('status-title').textContent = 'Item Reserved for You!';
        document.getElementById('status-message').textContent = 
          'Congratulations! The seller has accepted your request. This item is now reserved for you. Contact the seller to arrange pickup.';
      } else if (status === 'rejected') {
        step1.classList.add('completed');
        step2.classList.add('completed');
        progressFill.style.width = '100%';
        
        document.getElementById('status-icon').textContent = '❌';
        document.getElementById('status-title').textContent = 'Request Declined';
        document.getElementById('status-message').textContent = 
          'Unfortunately, the seller has declined your purchase request. You can browse other items or contact the seller for more information.';
      }
      
      // Show buyer note if exists
      if (purchase.note) {
        document.getElementById('buyer-note-section').style.display = 'block';
        document.getElementById('buyer-note').textContent = purchase.note;
      }
      
      // Show seller note if exists
      if (purchase.sellerNote) {
        document.getElementById('seller-note-section').style.display = 'block';
        document.getElementById('seller-note').textContent = purchase.sellerNote;
      }
    }

    function goToChat() {
      if (!currentPurchase || !currentItem) return;
      var sellerEmail = currentPurchase.sellerEmail;
      var sellerName = sellerEmail.split('@')[0];
      var itemImage = currentItem.images && currentItem.images[0] ? currentItem.images[0] : currentItem.image || '';
      
      window.location.href = 'chat.html?itemId=' + encodeURIComponent(currentPurchase.itemId) +
        '&otherEmail=' + encodeURIComponent(sellerEmail) +
        '&otherName=' + encodeURIComponent(sellerName) +
        '&itemName=' + encodeURIComponent(currentItem.name) +
        '&itemImage=' + encodeURIComponent(itemImage);
    }

    function goToItem() {
      if (!currentPurchase) return;
      window.location.href = 'product-details.html?id=' + encodeURIComponent(currentPurchase.itemId);
    }
